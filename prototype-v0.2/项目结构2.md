&nbsp;       currentGLBUrl = url;

&nbsp;       const loadTimeout = setTimeout(() => {

&nbsp;           showError('模型加载超时，请检查网络或模型URL是否有效');

&nbsp;                       obj.receiveShadow = true;

&nbsp;               const center = new THREE.Vector3();
